# События

# Как подключить

  1. Создать файл, например - campaign.js
  2. Скопировать скрипт ниже в этот файл
var campaign={route:"https://person-lists.herokuapp.com/",visitor:"",visitorInfo:null,init:function(visitorId,visitorInfo){this.visitorInfo=visitorInfo?visitorInfo:null;if(!visitorId){var userFromStore=localStorage.getItem("campaignVisitor");if(!userFromStore){this.visitor=this._createUUID()
localStorage.setItem("campaignVisitor",this.visitor);this._saveUser()}else{this.visitor=userFromStore;if(visitorInfo){this.updateVisitor(visitorInfo)}}}else{this.visitor=visitorId;if(visitorInfo){this.updateVisitor(visitorInfo)}}},setClicks:function(){var arr=arguments;var that=this;var i;for(i=1;i<arr.length;i++){that.detectClick(arr[i],arr[0])}},detectClick:function(el,type){var that=this;var domEl=null;document.addEventListener('DOMContentLoaded',function(){if(type==="class"){domEl=document.getElementsByClassName(el)[0]}else{domEl=document.getElementById(el)}
if(domEl){domEl.addEventListener('click',function(){that.triggerEvent(el)})}else{console.error("Campaign: DOM element is not found")}})},triggerEvent:function(trigger,productId){if(!trigger){console.error("Campaign: event is not defined")
return}
var xhr=new XMLHttpRequest();xhr.open("POST",this.route+'triggers/'+'?event='+trigger+'&visitor='+this.visitor);xhr.setRequestHeader('Content-type','application/json; charset=utf-8');if(productId){xhr.send(JSON.stringify({productId:productId}))}else{xhr.send()}},setVisitorId(id){if(this.visitor){this._changeVisitorId(id)}else{console.error("Campaign: Visitor id is not defined")}},updateVisitor(params){if(!this.visitor){console.log("Campaign: No visitor id is detected")}
var updateParams={};if(params.email){updateParams.email=params.email}
if(params.phone){updateParams.phone=params.phone}
var xhr=new XMLHttpRequest();xhr.open("PATCH",this.route+'visitors/'+this.visitor);xhr.setRequestHeader('Content-type','application/json; charset=utf-8');xhr.send(JSON.stringify(updateParams))},_createUUID:function(){return'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g,(c)=>{var r=Math.random()*16|0,v=c=='x'?r:(r&0x3|0x8);return v.toString(16)})},_saveUser(){var xhr=new XMLHttpRequest();xhr.open("POST",this.route+'visitors');xhr.setRequestHeader('Content-type','application/json; charset=utf-8');xhr.send(JSON.stringify({visitor:this.visitor,email:this.visitorInfo&&this.visitorInfo.email?this.visitorInfo.email:"",phone:this.visitorInfo&&this.visitorInfo.phone?this.visitorInfo.phone:"",}))},_changeVisitorId(id){var xhr=new XMLHttpRequest();xhr.open("PATCH",this.route+'visitors/'+this.visitor);xhr.setRequestHeader('Content-type','application/json; charset=utf-8');xhr.send(JSON.stringify({oldId:this.visitor,visitor:id}));this.visitor=id}}
  1. Подключить скрипт на каждую страницу, на которой нужно отслеживать события для campaign
  <script type="text/javascript" src="./js/campaign.js"></script>

WARNING

В src проставить правильный путь до файла

# Как использовать

После подключения скрипта сразу вызвать метод campaign.init:

  • первым параметром передается идентификатор клиента в campaign, если такого нет, то передавать null;
  • вторым параметром передается информация по клиенту, пока принимаются параметры - email и phone

# Пример инициализации

  // если нет никакой информации по клиенту - без параметров
  campaign.init();
  // если есть идентификатор клиента
  campaign.init("95849854");
  // если есть информация по клиенту, а идентификатора нет
  campaign.init(null, {email: "testfromweb@lflf.ru", phone: "+79638988555"});

# Пример использования

  // когда мы хотим вызвать напрямую метод совершения триггера
  // в параметре передать идентификатор события заведенного в campaign
  campaign.triggerEvent("test_btn");

  // если отслеживаем клик на странице по какому либо одному элементу
  // первый параметр - id или class элемента, клик по которому мы отслеживаем
  // второй параметр - если ищем элемент по классу - передаем строку - "class", если по id - ничего не передаем
  campaign.detectClick("about-product", "class");

  // если отслеживаем клик на странице по нескольким элементам
  // первый параметр - если ищем элементы по классам - передаем строку - "class", если по id - передаем null
  // все последующие параметры - это id или классы элементов
  campaign.setClicks(null, "about-product", "test-product", "banner-click");

WARNING

Чтобы триггер события отработал нужно обязательно завести событие с идентификатором в campaign. Если отслеживаем клики по элементам - идентификатор события - это класс или id элемента. Они должны быть уникальны.

# Полные примеры и Use cases

# Клик по элементу

Описание: Нужно отслеживать клик по одному элементу на странице

# Действия:

  1. Заводим событие в campaign
  2. Заполняем поле идентификатор, например - sale-product-01
  3. На странице с элементом, который отслеживаем, проставляем класс "sale-product-01" для элемента
    <button class="sale-product-01"></button>
  4. Добавляем скрипт на страницу, инициализируем campaign с информацией по клиенту и запускаем слушателя события
  <script type="text/javascript" src="./js/campaign.js"></script>
  campaign.init("50960596", {email: "testfromweb@lflf.ru", phone: "+79638988555"});
  campaign.detectClick("sale-product-01", "class");

# Добавление продукта в корзину

Описание: Хотим вызывать триггер события campaign в каком-либо методе на странице, например, при добавлении продукта в корзину

# Действия:

  1. Заводим событие в campaign
  2. Заполняем поле идентификатор, например - sale-product-01
  3. Добавляем скрипт на страницу, инициализируем campaign с информацией по клиенту
  <script type="text/javascript" src="./js/campaign.js"></script>
  campaign.init("50960596", {email: "testfromweb@lflf.ru", phone: "+79638988555"});
  1. После метода, который вызывается при добавлении продукта в корзину, вызывать метод campaign - triggerEvent
campaign.triggerEvent("sale-product-01", "58868686"); // второй параметр дополнительный, необязательный, сейчас передается productId

# Дополнительные методы

# Установить id клиента после инициализации

Описание: Хотим установить id клиента после инициализации campaign, например, до инициализации скрипта campaign (c методом campaign.init()) id клиента был неизвестен, а после пользователь авторизовался

campaign.setVisitorId('N4858585'); // в параметре передаем id пользователя

# Обновить информацию по клиенту после инициализации

Описание: Хотим обновить информацию о клиенте, например, до инициализации скрипта campaign (c методом campaign.init()) эта информация была неизвестна. Ниже пример с обновлением email

campaign.updateVisitor({ email: 'newemail@mail.ru'}); // передаем объект с новыми параметрами - на данный момент только параметры email и phone
